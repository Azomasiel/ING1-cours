CountingSort(A, n, k):
    for i <- 0 to k - 1:
        C[i] <- 0
    for i <- 0 to n - 1:
        C[A[i]] <- C[A[i]] + 1
    for i <- 1 to k - 1:
        C[i] <- C[i] + C[i - 1]
    for i <- n - 1 down to 0:
        C[A[i]] <- C[A[i]] - 1
        B[C[A[i]]] <- A[i]
    return B
