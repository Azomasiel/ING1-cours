Exemples concrets

void fun(int n)
{
    for (int i = 5; i <= n; ++i)
    {
        puts("A");
        for (int j = 10; j < 20; ++j)
            puts("B")
        for (int j = 1; j < i; ++j)
            puts("C");
        for (int j = 10; j < i; ++j)
            puts("D");
    }
}
