Selection(A, b, e, k)
// retourne la valeur de rang k dans A[b...e-1]
// 0 <= k < e - b
if (e - b) == 1
    return A[b]
else
    m <- Partition(A, b, e)
    l <- m - b
    if (k < l)
        return Selection(A, b, m, k)
    else
        return Selection(A, m, l, k - l)
